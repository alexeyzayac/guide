#### Пример Dockerfile **без ENTRYPOINT** (только CMD)
```dockerfile
#использовать образ
FROM python:3.9-slim
#запустить команду
RUN apt-get update && apt-get install -y --no-install-recommends curl 
#рабочий директорий
WORKDIR /app
#копируем файл в рабочий директорий
COPY hello.py .
#команда по умолчанию
CMD ["python", "hello.py"]
```
#### Пример Dockerfile **с ENTRYPOINT**
```dockerfile
FROM python:3.9-slim
WORKDIR /app
COPY hello.py .
#пробросить порт 3000
EXPOSE 3000
#фиксированная точка входа – интерпретатор и скрипт
ENTRYPOINT ["python", "hello.py"]
```
#### Пример многоступенчатой сборки (Go)
```bash
#этап сборки
FROM golang:1.21 AS builder
WORKDIR /src
COPY go.mod go.sum ./
RUN go mod download
COPY . .
RUN CGO_ENABLED=0 GOOS=linux go build -o myapp .

#финальный образ
FROM alpine:latest
RUN apk --no-cache add ca-certificates
WORKDIR /root/
COPY --from=builder /src/myapp .
ENTRYPOINT ["./myapp"]
CMD ["--help"]
```
